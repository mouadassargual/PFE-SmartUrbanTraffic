\section*{Commandes de lancement}

\textbf{Validation MDP — 5 scénarios :}

\begin{lstlisting}[language=bash]
python3 test_mdp.py
\end{lstlisting}

\textbf{Pipeline complet — mode analyse :}

\begin{lstlisting}[language=bash]
python3 -m pipeline.main \
  --video data/videos/ne8th/Bellevue_NE8th.mp4 \
  --model models/downloads/YOLO26n_step3_960_best.onnx \
  --imgsz 960 \
  --conf 0.35 \
  --dashboard \
  --person-roi \
  --vehicle-roi
\end{lstlisting}

\textbf{Pipeline complet — meilleur
compromis embarqué :}

\begin{lstlisting}[language=bash]
python3 -m pipeline.main \
  --video data/videos/ne8th/Bellevue_NE8th.mp4 \
  --model models/downloads/YOLO26n_step3_800_best.onnx \
  --imgsz 800 \
  --conf 0.35 \
  --dashboard \
  --person-roi \
  --vid-stride 2
\end{lstlisting}

\textbf{Journalisation MLflow :}

\begin{lstlisting}[language=bash]
python3 scripts/log_mlflow_final.py
\end{lstlisting}
